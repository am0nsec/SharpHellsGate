174  public bool GenerateRWXMemorySegment() {
175  // Find and JIT the method
176  MethodInfo method = typeof(HellsGate).GetMethod(nameof(Gate), BindingFlags.Static | BindingFlags.NonPublic);
177  if (method == null) {
178  Util.LogError("Unable to find the method");
179  return false;
180  }
181  RuntimeHelpers.PrepareMethod(method.MethodHandle);
182 
183  // Get the address of the function and check if first opcode == JMP
184  IntPtr pMethod = method.MethodHandle.GetFunctionPointer();
185  if (Marshal.ReadByte(pMethod) != 0xe9) {
186  Util.LogError("Method was not JIT'ed or invalid stub");
187  return false;
188  }
189  Util.LogInfo($"Managed method address: 0x{pMethod:x16}");
190 
191  // Get address of jited method and stack alignment
192  Int32 offset = Marshal.ReadInt32(pMethod, 1);
193  UInt64 addr = (UInt64)pMethod + (UInt64)offset;
194  while (addr % 16 != 0)
195  addr++;
196  Util.LogInfo($"Unmanaged method address: 0x{addr:x16}\n");
197 
198  this.MangedMethodAddress = method.MethodHandle.GetFunctionPointer();
199  this.UnmanagedMethodAddress = (IntPtr)addr;
200  this.IsGateReady = true;
201  return true;
202  }
